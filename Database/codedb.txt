Table users {
  user_id int [pk]
  username varchar
  email varchar
  password_hash varchar
  full_name varchar
  is_admin boolean
  created_at datetime
}

Table orders {
  order_id int [pk]
  user_id int [ref: > users.user_id]
  order_date datetime
  status varchar
  total_amount decimal
}

Table order_details {
  order_detail_id int [pk]
  order_id int [ref: > orders.order_id]
  game_id int [ref: > games.game_id]
  unit_price decimal
}

Table games {
  game_id int [pk]
  title varchar
  description text
  genre varchar
  platform varchar
  price decimal
  stock int
  image_url varchar
  created_at datetime
  modified_by int [ref: > users.user_id]
}

Table game_categories {
  game_id int [ref: > games.game_id]
  category_name varchar
  modified_by int [ref: > users.user_id]
}

Table news {
  news_id int [pk]
  title varchar
  content text
  image_url varchar
  published_at datetime
  modified_by int [ref: > users.user_id]
}
